    <script>
        // AI Assistant Toggle
        document.querySelectorAll('.ai-assistant').forEach(assistant => {
            assistant.addEventListener('click', function() {
                const aiType = this.getAttribute('data-ai');
                const message = document.getElementById(`${aiType}-message`);
                const chevron = this.querySelector('.fa-chevron-down');
                
                message.classList.toggle('active');
                chevron.classList.toggle('rotate-180');
            });
        });

        // Simulate AI updates
        function updateAIRecommendations() {
            const riskMessages = [
                "Risk exposure remains optimal at 1.8% per trade. No immediate threats detected.",
                "Warning: GBP/USD position showing elevated volatility - consider tightening stop-loss to 1.2x ATR.",
                "Market liquidity decreasing - reducing position sizes by 10% as precaution.",
                "All positions within safe risk parameters. Current drawdown at 4.2% of max allowed."
            ];
            
            const analysisMessages = [
                "Strong bullish momentum detected on USD/JPY with 82% confidence.",
                "MACD divergence suggests potential reversal on GBP/USD in next 2-4 hours.",
                "London session opening shows increased volume - favorable trading conditions.",
                "EUR/USD approaching key resistance at 1.1050 - watch for rejection patterns."
            ];
            
            const performanceMessages = [
                "Current win rate at 78.4% (+2.1% from yesterday).",
                "EMA crossover strategy performing exceptionally with 84% accuracy.",
                "Suggestion: Increase position size by 15% for high-confidence signals.",
                "Scalping strategy underperforming during Asian session - temporarily disabled."
            ];
            
            // Rotate through messages
            document.getElementById('risk-message').querySelector('p').textContent = riskMessages[Math.floor(Math.random() * riskMessages.length)];
            document.getElementById('analysis-message').querySelector('p').textContent = analysisMessages[Math.floor(Math.random() * analysisMessages.length)];
            document.getElementById('performance-message').querySelector('p').textContent = performanceMessages[Math.floor(Math.random() * performanceMessages.length)];
            
            // Randomly pulse active status
            document.querySelectorAll('.ai-assistant .bg-green-500').forEach(status => {
                if (Math.random() > 0.7) {
                    status.classList.add('pulse');
                    setTimeout(() => status.classList.remove('pulse'), 2000);
                }
            });
        }

        // Simulate live data updates
        function updateTradeData() {
            const profitElements = document.querySelectorAll('.trade-card .font-medium');
            profitElements.forEach(el => {
                if (el.textContent.includes('+') || el.textContent.includes('-')) {
                    const currentValue = parseFloat(el.textContent.match(/[\d.]+/)[0]);
                    const randomChange = (Math.random() * 5 - 2).toFixed(1);
                    const newValue = (currentValue + parseFloat(randomChange)).toFixed(1);
                    
                    if (newValue > currentValue) {
                        el.textContent = `+${newValue} pips`;
                        el.className = 'text-green-400 font-medium';
                    } else if (newValue < currentValue) {
                        el.textContent = `-${Math.abs(newValue)} pips`;
                        el.className = 'text-red-400 font-medium';
                    } else {
                        el.textContent = `${newValue} pips`;
                        el.className = 'text-gray-300 font-medium';
                    }
                }
            });

            // Update progress bars
            const progressBars = document.querySelectorAll('.progress-fill');
            progressBars.forEach(bar => {
                const currentWidth = parseFloat(bar.style.width) || 0;
                const randomChange = Math.random() * 10 - 5;
                let newWidth = currentWidth + randomChange;
                newWidth = Math.max(0, Math.min(newWidth, 100));
                bar.style.width = `${newWidth}%`;
                
                // Update color based on value
                if (newWidth > 70) {
                    bar.className = bar.className.replace(/bg-\w+-\d+/g, 'bg-green-500');
                } else if (newWidth > 30) {
                    bar.className = bar.className.replace(/bg-\w+-\d+/g, 'bg-yellow-500');
                } else {
                    bar.className = bar.className.replace(/bg-\w+-\d+/g, 'bg-red-500');
                }
            });
        }

        // Update every 3 seconds
        setInterval(updateTradeData, 3000);
        setInterval(updateAIRecommendations, 8000);

        // Simulate blinking for active elements
        setInterval(() => {
            const blinkElements = document.querySelectorAll('.bg-blue-900');
            blinkElements.forEach(el => {
                el.classList.toggle('bg-opacity-50');
                el.classList.toggle('bg-opacity-30');
            });
        }, 1000);

        // Initial AI update
        updateAIRecommendations();
    </script>
